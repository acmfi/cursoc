		\section {Sintaxis}
		\begin {enumerate}
			\item ¿Que falla en éste código?
			\begin {verbatim}
int main (void)
{
        printf ("Hola Mundo\n")
        return 0;
}
			\end{verbatim}
			\item ¿Que imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int a = 0, b = 1;
   if (a == 1)
      a++;
      b++;
   printf ("a = %d\n", a);
   printf ("b = %d\n", b);
   return 0;
}
			\end{verbatim}
			\item ¿Qué falla en éste código?
			\begin {verbatim}
int main (void)
{
   int a = 0, b = 1;
   if (a == 1)
      a++;
      b++;
   else
      b--;
   printf ("a = %d\n", a);
   printf ("b = %d\n", b);
   return 0;		
}
			\end{verbatim}
			\item ¿Qué falla en éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   for (i = 0, i < 10, i++)
      printf ("%d\n", i);
   return 0;
}
			\end{verbatim}
			\item ¿Qué falla en éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   for (i = 0; i < 10; i++)
      printf ("%d\n", I);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   for (i = 0; i < 10; i++);
      printf ("%d\n", i);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   for (i = 0; i > 10; i++) {
      printf ("%d\n", i);
      if (i == 4)
         break;
   }
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   for (i = 0; i < 10; i++) {
      if (i < 5)
         continue;
      printf ("%d\n", i);
   }
   return 0;
}
			\end{verbatim}
		\end {enumerate}

		\section {Punteros}
		\begin {enumerate}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int a = 0;
   int *ptr;
   
   ptr = &a;
   a = a + 3;
   printf ("%d", *ptr);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int x = 1, y = 2;
   int *ptr;
   
   ptr = &x;
   y = *ptr;
   *ptr = 0;
   printf ("%d", *ptr);
   printf ("%d", x);
   printf ("%d", y);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int x = 1, y = 2;
   int *ptr_x, *ptr_y;
   
   ptr_x = &x;
   ptr_y = &y;
   ++*ptr_x;
   ptr_x = ptr_y;
   printf ("%d", *ptr_x);
   printf ("%d", *ptr_y);
   printf ("%d", x);
   printf ("%d", y);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
void incrementar (int a)
{
   a = a + 1;
}

int main (void)
{
   int i, a = 0;
   
   for (i = 0; i < 10; i++)
      incrementar (a);
   printf ("%d", a);
   return 0;
}
			\end{verbatim}
			\item ¿Qué falla en éste código?
			\begin {verbatim}
void incrementar (int *a)
{
   a = a + 1;
}

int main (void)
{
   int i, a = 0;
   
   for (i = 0; i < 10; i++)
      incrementar (&a);
   printf ("%d", a);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
void imprimir (int a)
{
   a = a + 1;
   printf ("%d\n", a);
}

int main (void)
{
   int i, a = 0;
   for (i = 0; i < 7; i++)
      imprimir (a);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
void imp_inc (int *a)
{
   *a = *a + 1;
   printf ("%d\n", *a);
}

int main (void)
{
   int i, a = 0;
   for (i = 0; i < 5; i++)
      imprimir (&a);
   return 0;
}
			\end{verbatim}
		\end {enumerate}

		\section {Arrays}
		\begin {enumerate}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int i, j = 0;
   int lista [10];
   
   for (i = 1; i < 10; i++, j++)
      lista [i] = j++;
   for (i = 5; i < 10; i++)
      printf ("%d\n", lista [i]);
   return 0;
}
			\end{verbatim}
			\item ¿Qué falla en éste código?
			\begin {verbatim}
int main (void)
{
   int i;
   int lista [10];
   
   while (i < 10)
      printf ("%d\n", lista [i++]);
   return 0;
}
			\end{verbatim}
			\item ¿Qué imprimiría éste código?
			\begin {verbatim}
int main (void)
{
   int i, a;
   int lista [5];
   i = a = 0;
   for (; i < 5; ) {
      printf ("%d", ++a);
      printf ("%d\n", lista [i++]);
   }
   return 0;
}
			\end{verbatim}
		\end {enumerate}

\section{Test: Punteros}

\begin{verbatim}
int *pnumero;
int num1,num2;
char *pchar;
char letra1;
num1 = 2;
num2 = 5;
letra1 = 'a';
\end{verbatim}

\begin{enumerate}
\item \texttt{pnumero = \&num1;} \\ Valor de \texttt{pnumero}:
\begin{enumerate}
\item $2$.
\item La dirección de memoria que ocupa \texttt{pnumero}.
\item La dirección de memoria que ocupa \texttt{num1}.
\end{enumerate}

\item \texttt{num2 = num1 + *pnumero;} \\ Valor de \texttt{num2}:
\begin{enumerate}
\item $7$.
\item $4$.
\item La dirección de \texttt{pnumero} más 2.
\end{enumerate}

\item
\begin{verbatim}
pchar = &letra1;
*pchar = 'B';
\end{verbatim}
Valor de \texttt{\&pchar}:
\begin{enumerate}
\item Dirección de memoria de \texttt{pchar}.
\item ``a''.
\item ``B''.
\end{enumerate}

\item ¿Cómo obtengo el valor de \texttt{letra1}?
\begin{enumerate}
\item Con \texttt{\&letra1} o \texttt{*pchar}.
\item Con \texttt{letra1} o   \texttt{*pchar}.
\item Con \texttt{*letra1} o  \texttt{\&pchar}.
\end{enumerate}

\item ¿Qué pasos efectua \\ \texttt{num2 = * ((int *) pchar);}?
\begin{enumerate}
\item Desreferencia \texttt{pchar}, lo convierte a \texttt{int} y lo asigna a
\texttt{num2}.
\item Convierte \texttt{pchar} a un puntero a \texttt{int}, lo desreferencia y
lo asigna a \texttt{num2}.
\item Obtiene la dirección de \texttt{pchar} como si fuera un \texttt{int} y lo
asigna a \texttt{num2}.
\end{enumerate}

\end{enumerate}

\section{Test: Cadenas}

\begin{verbatim}
typedef struct nodo * cadena;
typedef struct nodo {
    int dato;
    cadena siguiente;
} nodo;
cadena micadena;
nodo   minodo;
nodo   tunodo;
\end{verbatim}

\begin{enumerate}
\item Elige la instrucción más correcta:
\begin{enumerate}
\item \texttt{micadena = *minodo;}
\item \texttt{*micadena = minodo;}
\item \texttt{micadena = \&minodo;}
\end{enumerate}

\item Elige la instrucción más correcta:
\begin{enumerate}
\item \texttt{minodo.siguiente = tunodo;}
\item \texttt{minodo.siguiente = \&tunodo;}
\item \texttt{minodo.siguiente = *tunodo;}
\end{enumerate}

\item Suponiendo que \texttt{micadena} apunta a un nodo, elige la instrucción más correcta:
\begin{enumerate}
\item \texttt{micadena.dato  = 4;}
\item \texttt{micadena->dato = 4;}
\item \texttt{micadena.(*dato) = 4;}
\end{enumerate}

\item Suponiendo que \texttt{micadena} apunta a un nodo, elige la instrucción más correcta:
\begin{enumerate}
\item \texttt{minodo.siguiente = micadena->cadena;}
\item \texttt{minodo.siguiente = *micadena;}
\item \texttt{\&(minodo.siguiente) = micadena;}
\end{enumerate}

\item ¿Qué efecto produce \texttt{minodo.siguiente = \&micadena}?
\begin{enumerate}
\item Asignación de \texttt{micadena} al nodo \texttt{minodo}.
\item Asignación de la dirección de \texttt{micadena} a la cadena del nodo \texttt{minodo}.
\item Asignación de \texttt{micadena} a la cadena del nodo \texttt{minodo}.
\end{enumerate}

\end{enumerate}
